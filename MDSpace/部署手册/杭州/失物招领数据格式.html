@ApiModelProperty("失物名称")
private String lostName;
@ApiModelProperty("失物编号")
private String lostCode;
@ApiModelProperty("失物分类")
private String lostType;
@ApiModelProperty("拾捡地点")
private String lostLocation;
@ApiModelProperty("失物描述")
private String description;
@ApiModelProperty("失物照片id")
private List<ImageFileDto> lostImageList;

// 拾捡人信息
@ApiModelProperty("拾捡人姓名")
private String pickerName;
@ApiModelProperty("拾捡人联系方式")
private String pickerPhone;
@ApiModelProperty("拾捡人身份")
private String pickerIdentity;
@ApiModelProperty("拾捡人部门")
private String pickerDept;
@ApiModelProperty("拾捡时间")
@JsonFormat(pattern = "yyyy-MM-dd HH:mm:ss", timezone = "GMT+8")
private Date pickTime;
@ApiModelProperty("安检送查人")
private String securitySendPerson;
@ApiModelProperty("安检检查人")
private String securityCheckPerson;

// 失物设置
@ApiModelProperty("是否向旅客公示")
private Boolean publicity;
@ApiModelProperty("是否公示失物照片")
private Boolean publicImage;
@ApiModelProperty("同步民航失物系统")
private Boolean syncOfficeSystem;
@ApiModelProperty("是否设置有效期")
private Boolean validDateEnable;
@ApiModelProperty("有效期")
@JsonFormat(pattern = "yyyy-MM-dd HH:mm:ss", timezone = "GMT+8")
private Date validTime;

@ApiModelProperty("上报人员Id")
private String userId;
@ApiModelProperty("状态 0:暂存 1:已登记 2:已入库 3:待认领 4:已认领")
private Integer status;

// 存储信息
@ApiModelProperty("存储位置")
private String storeLocation;
@ApiModelProperty("失物照片id")
private List<ImageFileDto> storeImageList;
@ApiModelProperty("接收人Id")
private String storePersonId;
@ApiModelProperty("接收人")
private String storePerson;
@ApiModelProperty(value = "入库时间")
@JsonFormat(pattern = "yyyy-MM-dd HH:mm:ss", timezone = "GMT+8")
private Date storeTime;

// 认领信息
@ApiModelProperty("认领人姓名")
private String claimManName;
@ApiModelProperty("认领人联系方式")
private String claimManPhone;
@ApiModelProperty("证件号")
private String claimCardNo;
@ApiModelProperty("证件地址")
private String claimAddress;
@ApiModelProperty("备注")
private String claimDesc;
@ApiModelProperty(value = "认领日期")
@JsonFormat(pattern = "yyyy-MM-dd HH:mm:ss", timezone = "GMT+8")
private Date claimTime;
@ApiModelProperty("认领处理人")
private String claimUser;

// 完成认领
@ApiModelProperty("认领说明")
private String claimFinishDesc;
@ApiModelProperty("认领人签字文件")
private String claimSignFile;
@ApiModelProperty("认领人授权文件")
private String claimPowerFile;

@ApiModelProperty("移交人")
private String handoverPerson;
