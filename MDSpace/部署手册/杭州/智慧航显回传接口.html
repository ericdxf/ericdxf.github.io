public String id;

public String description;

@ApiModelProperty("是否在线 true表示在线  false表示离线")
public Boolean isOnline;

@ApiModelProperty("更新时间 更新时间用来判断是否在线")

public Date updateTime;

@ApiModelProperty("创建时间")
